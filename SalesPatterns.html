ANOVA Test¶

from scipy import stats

# Preparing data for ANOVA
data_anova = df[['Ship To Country', 'Sale Price', 'Profit']].dropna()

# Performing ANOVA for Sale Price
anova_sale_price = stats.f_oneway(
    *[group['Sale Price'].values for name, group in data_anova.groupby('Ship To Country')]
)

# Performing ANOVA for Profit
anova_profit = stats.f_oneway(
    *[group['Profit'].values for name, group in data_anova.groupby('Ship To Country')]
)

anova_results = {
    'ANOVA Sale Price': anova_sale_price,
    'ANOVA Profit': anova_profit
}

from mlxtend.frequent_patterns import apriori, association_rules

# Assuming the dataset has columns 'transaction_id' and 'item'
# Create a basket matrix
basket = df.groupby(['OrderId', 'Commodity'])['Commodity'].count().unstack().fillna(0)
basket = basket.applymap(lambda x: 1 if x > 0 else 0)

# Apply Apriori algorithm
frequent_itemsets = apriori(basket, min_support=0.01, use_colnames=True)

# Generate association rules
rules = association_rules(frequent_itemsets, metric="lift", min_threshold=1)

data['Date'] = data['Order Date'].dt.to_period('M').dt.to_timestamp()

# Count orders per month
